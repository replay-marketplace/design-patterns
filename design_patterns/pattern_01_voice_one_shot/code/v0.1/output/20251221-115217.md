## 2. List Processor (Sorting and Filtering)

```python
def sort_list(items):
    """Sort a list in ascending order."""
    result = sorted(items)
    print(f"[list_utils] Sorted {items} -> {result}")
    return result

def filter_even(items):
    """Filter out odd numbers, keep only even."""
    result = [x for x in items if x % 2 == 0]
    print(f"[list_utils] Filtered even from {items} -> {result}")
    return result

def sum_list(items):
    """Sum all items in a list."""
    result = sum(items)
    print(f"[list_utils] Sum of {items} = {result}")
    return result
```
